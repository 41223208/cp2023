      // 直線斜率
 
      double m = (y2 - y1) / (x2 - x1);
 
 
 
      // 直線方程式中的常數項
 
      double b = y1 - m * x1;
 
 
 
      // 圓與直線交點的計算
 
      double A = 1 + pow(m, 2);
 
      double B = 2 * (m * b - m * k - h);
 
      double C = pow(k, 2) - pow(r, 2) + pow(h, 2) - 2 * b * k + pow(b, 2);

    double discriminant = pow(B, 2) - 4 * A * C;
 
    if (discriminant > 0) {

        double y_intersect1 = m * x_intersect1 + b;
